export const recordAudio = () => {
  const recognition = new window.webkitSpeechRecognition();
  recognition.continuous = false;
  recognition.lang = 'en-US';

  return new Promise((resolve) => {
    recognition.start();
    recognition.onresult = (event) => {
      resolve(event.results[0][0].transcript);
    };
  });
};
export async function talkToAIFriend(message) {
  const res = await fetch(`${process.env.REACT_APP_API_URL}/api/ai/talk`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ message }),
  });
  return await res.json();
}
import OpenAI from 'openai';
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

export const talkToAI = async (req, res) => {
  const { message } = req.body;
  try {
    const completion = await openai.chat.completions.create({
      model: 'gpt-4-turbo',
      messages: [
        { role: 'system', content: 'You are a friendly English-speaking tutor.' },
        { role: 'user', content: message }
      ]
    });

    const reply = completion.choices[0].message.content;
    res.json({ reply });
  } catch (err) {
    res.status(500).json({ error: 'AI error', details: err });
  }
};

## Tech Stack
- **Frontend:** React + Web Speech API
